    Priorite des operateurs en c++

    Opérateur Sens d'évaluation
    () De gauche à droite
    ++, --, ! De droite à gauche
    *, /, % De gauche à droite
    +, - De gauche à droite
    <, <=, >, >= De gauche à droite
    ==, != De gauche à droite
    && De gauche à droite, s'arrête si faux
    || De gauche à droite, s'arrête si vrai
    ?: De droite à gauche
    =, +=, -=, *=, /=, %= De droite à gauche
    , De gauche à droite, retourne la valeur de droite
    Exemples de programmes c++
    Premier programme

    Ceci est un exemple simple de programme C++ affichant "Hello, World!" à l'écran.

                        
                        #include <iostream>
    
                        int main() {
                            cout << "Bonjour le monde!" << endl;
                            return 0;
                        }
    Variables et Opérations

    Ce programme utilise des variables et effectue quelques opérations de base.

                        
                        #include <iostream>
    
                        int main() {
                            // declaration des variables en c++
                            int a = 5;
                            int y = 20;
                            double b = 2.5;
                            char c = 'A';
    
                            cout << "a + b = " << a + b << endl;
                            cout << "Character: " << c << endl;
    
                            return 0;
                        }
    Conditions

    Ce programme utilise des structures conditionnelles pour afficher si un nombre est positif, négatif ou nul.

                        
                        #include <iostream>
    
                        int main() {
                            int num;
    
                            cout << "Entrez un nombre : ";
                            cin >> num;
    
                            if (num > 0) {
                                cout << "Le nombre est positif." << endl;
                            } else if (num < 0) {
                                cout << "Le nombre est négatif." << endl;
                            } else {
                                cout << "Le nombre est nul." << endl;
                            }
    
                            return 0;
                        }
    Boucles

    Ce programme utilise des boucles for, while et do...while pour afficher des itérations.

                        
                        #include <iostream>
    
                        int main() {
                            // Boucle For
                            for (int i = 0; i < 5; ++i) {
                                cout << "For loop iteration " << i << endl;
                            }
    
                            // Boucle While
                            int j = 0;
                            while (j < 5) {
                                cout << "While loop iteration " << j << endl;
                                ++j;
                            }
    
                            // Boucle Do...While
                            int k = 0;
                            do {
                                cout << "Do...While loop iteration " << k << endl;
                                ++k;
                            } while (k < 5);
    
                            return 0;
                        }
    Tableaux

    Ce programme utilise des tableaux uni-dimensionnels et multi-dimensionnels.

                        
                        #include <iostream>
    
                        int main() {
                            // Tableau uni-dimensionnel
                            int arr1[5] = {1, 2, 3, 4, 5};
    
                            // Tableau multi-dimensionnel
                            int arr2[2][3] = {{1, 2, 3}, {4, 5, 6}};
    
                            cout << "Element at index 2 in arr1: " << arr1[1] << endl;
                            cout << "Element at index (1,2) in arr2: " << arr2[1][2] << endl;
    
                            return 0;
                        }
    Fonctions

    Ce programme définit une fonction pour additionner deux nombres.

                        
                        #include <iostream>
    
                        // Déclaration de fonction
                        int add(int a, int b) {
                            return a + b;
                        }
    
                        int main() {
                            int result = add(3, 4);
                            cout << "3 + 4 = " << result << endl;
    
                            return 0;
                        }
